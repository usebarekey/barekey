## Multi-line snippets

```ts name="load-post.ts"
import { Effect } from "effect";
import { GetPost } from "$/docs/[category]/[slug]/page.remote";

export const load_post = ({ category, slug }: { category: string; slug: string }) =>
	Effect.gen(function* () {
		return yield* GetPost({ category, slug });
	});
```

```ser name="+page.svelte" icon="svelte"
<script lang="ts" effect>
  import { GetPost } from "$/docs/[category]/[slug]/page.remote";

  const { params } = $props();
  const category = $derived(params.category);
  const slug = $derived(params.slug);

  const post = yield* GetPost({
    category,
    slug
  });
</script>

<article>
  <h1>{post.frontmatter?.title}</h1>
  {@html post.html}
</article>
```
